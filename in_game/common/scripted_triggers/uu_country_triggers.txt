
REPLACE:alliance_can_exist = {
	scope:actor = {
		OR = {		#UU
			opinion = { target = scope:recipient value >= 0 }
			is_ai = no	
		}
		not = { has_blocked_treaties = scope:recipient }
		not = { is_subject_of = scope:recipient }
		not = { is_rival_of = scope:recipient }
		modifier:allowed_alliance = yes
		NOT = {
			is_member_of_international_organization_of_type = {
				type = coalition
				target = scope:recipient
			}
		}
		NOT = {
			in_union_with = scope:recipient
		}
		is_rebel_country = no
	}
	scope:recipient = {
		OR = {		#UU
			opinion = { target = scope:actor value >= 0 }
			is_ai = no	
		}
		not = { has_blocked_treaties = scope:actor }
		not = { is_subject_of = scope:actor }
		not = { is_rival_of = scope:actor }
		modifier:allowed_alliance = yes
		NOT = {
			is_member_of_international_organization_of_type = {
				type = coalition
				target = scope:actor
			}
		}
		is_rebel_country = no
	}
	can_create_subject_alliance = {
		actor = scope:actor
		recipient = scope:recipient
	}
	NOT = {
		 AND = {
			scope:actor = {
				is_hegemon = yes
			}
			scope:recipient = {
				is_hegemon = yes
			}
		}
	}
}

REPLACE:can_upgrade_country_rank_empire = {
	is_great_power = yes
	trigger_if = {
		limit = { 
			country_exists = c:PAP
			NOT = { tag = PAP }
			religion = religion:catholic 
		}
		c:PAP = {
			opinion = { target = root value > 100 }
		}
	}
	OR = {
		total_population > 10000
		total_true_faith_population > 7500
		total_accepted_culture_population > 5000
	}
	prestige > 35
	is_subject = no
}
REPLACE:can_upgrade_country_rank_kingdom = {
	OR = {
		total_population > 2000
		total_true_faith_population > 1500
		total_accepted_culture_population > 1000
		AND = {
			has_game_rule = rank_restriction_by_numbers
			is_member_of_international_organization = international_organization:hre
			OR = {
				total_population > 2000
				total_true_faith_population > 1500
				total_accepted_culture_population > 1000
			}
			international_organization:hre = {
				any_international_organization_member = {
					country_rank_level = 3
					count < 6
				}
			}
		}
	}
	prestige > 25		
	is_subject = no
}

		total_accepted_culture_population > 100
	}
