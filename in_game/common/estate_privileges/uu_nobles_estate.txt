overextension = {
	estate = nobles_estate

	potential = {
		NOT = {
			has_variable = overextension_revoked
		}
		OR = {
			has_or_had_tag = ENG
			has_or_had_tag = CAS
			has_or_had_tag = MAM
			has_or_had_tag = HUN
			has_or_had_tag = BOH
			has_or_had_tag = NAP
		}
	}

	allow = {
		NOT = {
			has_variable = overextension_revoked
		}
	}

	on_deactivate ={
		set_variable = {
			name = overextension_revoked
			value = yes
			hidden = yes
		}
	}


	country_modifier = {
		global_nobles_estate_power = 2.0
		nobles_estate_target_satisfaction = 0.1
		clergy_estate_target_satisfaction = -0.1
		peasants_estate_max_tax	= -0.05
		burghers_estate_max_tax = -0.05
		discipline = -0.05
	}
}
