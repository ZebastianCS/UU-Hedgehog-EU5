zmc_used_building_slots_percentage = { #building cap mapmode
  value = 1
  subtract = {
    value = modifier:free_building_levels
    subtract = total_building_levels
    divide = {
      value = modifier:free_building_levels
      min = 1
    }
  }
}

zmc_local_monthly_development = { #dev growth mapmode
  value = modifier:local_monthly_development
  if = {
  	limit = { exists = owner }
  	add = owner.modifier:global_monthly_development
  }
  multiply = {
    value = 1
    add = modifier:local_monthly_development_modifier
  multiply = 100
  }
}





zmc_time_until_100_development = { #time until 100 dev mapmode
  value = 100 #100 development
  subtract = development #subtract current development
  divide = zmc_time_until_100_development_sub_value
  divide = 12 #convert from months to years
}

zmc_time_until_100_development_sub_value = {
  value = modifier:local_monthly_development
  if = {
  	limit = { exists = owner }
  	add = owner.modifier:global_monthly_development
  }
  multiply = {
    value = 1
    add = modifier:local_monthly_development_modifier
  }
  min = 0.0001 #to avoid dividing by 0
}

zmc_pop_cap = {
  value = modifier:local_population_capacity
  multiply = 1000 #1000 #for 1 pop being 1000 people
  multiply = zmc_pop_cap_temp_value
}

zmc_pop_cap_temp_value = {
  value = modifier:local_population_capacity_modifier
  add = 1 #because in game these modifiers are defined as 0.1 meaning 110%
}

  value = population

zmc_local_max_literacy = { #max literacy mapmode
	every_pop = {
		limit = { pop_size > 0 }
		add = {
			if = {
				limit = { pop_type = pop_type:nobles }
				add = prev.modifier:local_nobles_max_literacy
				add = owner.modifier:global_nobles_max_literacy
			}
			else_if = {
				limit = { pop_type = pop_type:clergy }
				add = prev.modifier:local_clergy_max_literacy
				add = owner.modifier:global_clergy_max_literacy
			}
			else_if = {
				limit = { pop_type = pop_type:burghers }
				add = prev.modifier:local_burghers_max_literacy
				add = owner.modifier:global_burghers_max_literacy
			}
			else_if = {
				limit = { pop_type = pop_type:burghers }
				add = prev.modifier:local_burghers_max_literacy
				add = owner.modifier:global_burghers_max_literacy
			}
			else_if = {
				limit = { pop_type = pop_type:laborers }
				add = prev.modifier:local_laborers_max_literacy
				add = owner.modifier:global_laborers_max_literacy
			}
			else_if = {
				limit = { pop_type = pop_type:soldiers }
				add = prev.modifier:local_soldiers_max_literacy
				add = owner.modifier:global_soldiers_max_literacy
			}
			else_if = {
				limit = { pop_type = pop_type:peasants }
				add = prev.modifier:local_laborers_max_literacy
				add = owner.modifier:global_peasants_max_literacy
			}
			else_if = {
				limit = { pop_type = pop_type:tribesmen }
				add = prev.modifier:local_tribesmen_max_literacy
				add = owner.modifier:global_tribesmen_max_literacy
			}
			else_if = {
				limit = { pop_type = pop_type:slaves }
				add = prev.modifier:local_slaves_max_literacy
				add = owner.modifier:global_slaves_max_literacy
			}
			add = prev.modifier:local_max_literacy
			add = owner.modifier:global_max_literacy
			multiply = {
				add = pop_size
				multiply = 1000
				divide = prev.population
			}
			divide = 1000
		}
	}
}
